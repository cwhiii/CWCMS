PAGES

CREATE TABLE `cwholema_playground`.`pages` ( `p_id` SMALLINT(255) NOT NULL AUTO_INCREMENT , `b_id` SMALLINT(255) NULL DEFAULT NULL , `name` TEXT NULL DEFAULT NULL , `url` TEXT NULL DEFAULT NULL , `content` TEXT NULL DEFAULT NULL , `assembled` TEXT NULL DEFAULT NULL , `private` BOOLEAN NOT NULL DEFAULT TRUE , `draft` BOOLEAN NOT NULL DEFAULT TRUE , `created` DATETIME NULL DEFAULT NULL , `updated` DATETIME NULL DEFAULT NULL , UNIQUE `p_id_index` (`p_id`)) ENGINE = InnoDB;


---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

BOOKS

CREATE TABLE `cwholema_playground`.`books` ( `b_id` SMALLINT(255) NOT NULL AUTO_INCREMENT , `name` TEXT NULL DEFAULT NULL , `created` DATETIME NOT NULL , `updated` DATETIME NULL , PRIMARY KEY (`b_id`)) ENGINE = InnoDB;


---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

SYTLES

CREATE TABLE `cwholema_playground`.`styles` ( `s_id` SMALLINT(255) NOT NULL AUTO_INCREMENT , `content` TEXT NULL DEFAULT NULL , `created` DATETIME NOT NULL , `updated` DATETIME NULL , PRIMARY KEY (`s_id`)) ENGINE = InnoDB;


---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------


IMAGES

CREATE TABLE `cwholema_playground`.`images` ( `i_id` INT(255) NOT NULL AUTO_INCREMENT , `url` TEXT NULL DEFAULT NULL , `image` BLOB NULL DEFAULT NULL , `created` DATETIME NULL , `updated` DATETIME NOT NULL , PRIMARY KEY (`i_id`)) ENGINE = InnoDB;


---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------

FLECKS

CREATE TABLE `cwholema_playground`.`flecks` ( `f_id` INT NOT NULL AUTO_INCREMENT , `name` TEXT NULL DEFAULT NULL , `content` TEXT NULL DEFAULT NULL , `draft` BOOLEAN NOT NULL DEFAULT TRUE , `updated` DATETIME NULL DEFAULT NULL , UNIQUE `f_id_index` (`f_id`)) ENGINE = InnoDB;
